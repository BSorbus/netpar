def put_dirty_individual(dirty_individual, cert, cust)
  individual = Individual.new()

  individual.number = dirty_individual.number
  individual.date_of_issue = dirty_individual.date_of_issue
  individual.valid_thru = dirty_individual.valid_thru
  individual.license_status = dirty_individual.license_status
  individual.application_date = dirty_individual.application_date

  individual.call_sign = dirty_individual.call_sign
  individual.category = dirty_individual.category
  individual.transmitter_power = dirty_individual.transmitter_power
  individual.certificate_number = dirty_individual.certificate_number
  individual.certificate_date_of_issue = dirty_individual.certificate_date_of_issue
  individual.certificate = cert if cert.present? # !!!
  individual.payment_code = dirty_individual.payment_code
  individual.payment_date = dirty_individual.payment_date
#  individual.payment = dirty_individual.
  individual.customer = cust  # !!!
  individual.note = dirty_individual.note
  individual.user = dirty_individual.user
  puts " - zapisałem pozwolenie"

  dirty_station = DirtyAddress.find_by(code_table: 'A', code_table_id: dirty_individual.code)
  individual.station_city = dirty_station.address_city
  individual.station_street = dirty_station.address_street
  individual.station_house = dirty_station.address_number
  individual.station_number = dirty_station.address_house
  puts " - zapisałem lokalizacje stacji"

  individual.save


  puts " "
  puts " - usunalem lokalizacje stacji z DirtyAddress"
  DirtyAddress.where(code_table: 'A', code_table_id: dirty_individual.code).destroy_all
  puts " - usunalem pozwolenie z DirtyIndividual: #{dirty_individual.number}"

  dirty_individual.destroy        
end
